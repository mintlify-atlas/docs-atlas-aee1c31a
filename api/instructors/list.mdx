---
title: "List Instructors"
api: "GET /api/v1/instructores"
---

Retrieves a list of all instructors in the riding school system.

## Response

<ResponseField name="instructors" type="array">
  Array of instructor objects
  
  <Expandable title="Instructor object">
    <ResponseField name="id" type="long">
      Unique identifier for the instructor
    </ResponseField>
    
    <ResponseField name="dni" type="string">
      National identification number (7-9 digits)
    </ResponseField>
    
    <ResponseField name="nombre" type="string">
      First name of the instructor
    </ResponseField>
    
    <ResponseField name="apellido" type="string">
      Last name of the instructor
    </ResponseField>
    
    <ResponseField name="nombreCompleto" type="string">
      Full name (nombre + apellido)
    </ResponseField>
    
    <ResponseField name="fechaNacimiento" type="date">
      Birth date in ISO 8601 format (YYYY-MM-DD)
    </ResponseField>
    
    <ResponseField name="telefono" type="string">
      Phone number (10-15 characters, may include + prefix)
    </ResponseField>
    
    <ResponseField name="email" type="string">
      Email address
    </ResponseField>
    
    <ResponseField name="activo" type="boolean">
      Whether the instructor is currently active
    </ResponseField>
    
    <ResponseField name="color" type="string">
      Assigned color code for the instructor (hex format, max 7 characters)
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.ridingschool.example/api/v1/instructores" \
  -H "Content-Type: application/json"
```
</RequestExample>
